# Adobe Commerce Dropin Slots Reference

Slots allow you to inject custom HTML content into specific areas of dropin containers without changing their internal structure.

## How Slots Work

```javascript
import { render } from '@dropins/storefront-cart/render.js';
import { CartSummaryList } from '@dropins/storefront-cart/containers/CartSummaryList.js';

render(CartSummaryList, {
  slots: {
    SlotName: (ctx) => {
      // ctx contains data specific to that slot
      const element = document.createElement('div');
      element.innerHTML = 'Your custom content';
      return element;
    }
  }
})(document.getElementById('container'));
```

## Examples

### Custom Cart Thumbnail with Badge

```javascript
render(CartSummaryList, {
  slots: {
    Thumbnail: ({ item, defaultImageProps }) => {
      const div = document.createElement('div');
      div.className = 'custom-thumbnail';
      div.innerHTML = `
        <img src="${defaultImageProps.src}" alt="${defaultImageProps.alt}" />
        ${item.quantity > 1 ? `<span class="qty-badge">${item.quantity}</span>` : ''}
      `;
      return div;
    }
  }
})(element);
```

### Custom Empty Cart Message

```javascript
render(MiniCart, {
  slots: {
    EmptyCart: () => {
      const div = document.createElement('div');
      div.innerHTML = `
        <div style="text-align: center; padding: 40px;">
          <p>Your bag is empty!</p>
          <a href="/shop">Start Shopping</a>
        </div>
      `;
      return div;
    }
  }
})(element);
```

### Custom Product Actions (Add to Cart Area)

```javascript
render(ProductDetails, {
  sku: 'ABC123',
  slots: {
    Actions: (ctx) => {
      const { data, valid, appendButton } = ctx;

      // Add custom button
      appendButton({
        text: 'Add to Wishlist',
        icon: 'Heart',
        variant: 'secondary',
        onClick: () => console.log('Added to wishlist')
      });

      return null; // Return null to keep default + your additions
    }
  }
})(element);
```

### Custom Payment Method

```javascript
render(PaymentMethods, {
  slots: {
    Methods: {
      'checkmo': {
        enabled: true,
        displayLabel: true,
        icon: '/icons/check.svg',
        render: ({ cartId, replaceHTML }) => {
          const div = document.createElement('div');
          div.innerHTML = `
            <p>Pay by check or money order</p>
            <p>Mail to: 123 Commerce St</p>
          `;
          replaceHTML(div);
          return div;
        }
      }
    }
  }
})(element);
```
